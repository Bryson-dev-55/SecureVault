import React, { useState, useEffect } from 'react';
import { Lock, Unlock, Eye, EyeOff, FileText, Mail, Image, Key, Plus, Trash2, Download, X, File } from 'lucide-react';

export default function SecureVault() {
  const [isLocked, setIsLocked] = useState(true);
  const [password, setPassword] = useState('');
  const [error, setError] = useState('');
  const [activeTab, setActiveTab] = useState('passwords');
  const [showPassword, setShowPassword] = useState({});
  const [fullscreenImage, setFullscreenImage] = useState(null);
  
  const [passwords, setPasswords] = useState([]);
  const [emails, setEmails] = useState([]);
  const [documents, setDocuments] = useState([]);
  const [images, setImages] = useState([]);
  const [files, setFiles] = useState([]);
  
  const [showAddForm, setShowAddForm] = useState(false);
  const [newItem, setNewItem] = useState({ title: '', content: '', username: '' });

  useEffect(() => {
    if (!isLocked) {
      loadData();
    }
  }, [isLocked]);

  const loadData = () => {
    const savedPasswords = JSON.parse(localStorage.getItem('vault_passwords') || '[]');
    const savedEmails = JSON.parse(localStorage.getItem('vault_emails') || '[]');
    const savedDocuments = JSON.parse(localStorage.getItem('vault_documents') || '[]');
    const savedImages = JSON.parse(localStorage.getItem('vault_images') || '[]');
    const savedFiles = JSON.parse(localStorage.getItem('vault_files') || '[]');
    
    setPasswords(savedPasswords);
    setEmails(savedEmails);
    setDocuments(savedDocuments);
    setImages(savedImages);
    setFiles(savedFiles);
  };

  const saveData = (type, data) => {
    localStorage.setItem(`vault_${type}`, JSON.stringify(data));
  };

  const handleUnlock = (e) => {
    e.preventDefault();
    if (password === 'PASSWORD') {
      setIsLocked(false);
      setError('');
      setPassword('');
    } else {
      setError('Incorrect password!');
    }
  };

  const handleLock = () => {
    setIsLocked(true);
    setPassword('');
    setShowAddForm(false);
  };

  const handleAddItem = () => {
    if (!newItem.title) return;
    
    const item = {
      id: Date.now(),
      ...newItem,
      dateAdded: new Date().toLocaleDateString('en-US')
    };

    let updatedData;
    switch(activeTab) {
      case 'passwords':
        updatedData = [...passwords, item];
        setPasswords(updatedData);
        saveData('passwords', updatedData);
        break;
      case 'emails':
        updatedData = [...emails, item];
        setEmails(updatedData);
        saveData('emails', updatedData);
        break;
      case 'documents':
        updatedData = [...documents, item];
        setDocuments(updatedData);
        saveData('documents', updatedData);
        break;
      case 'images':
        updatedData = [...images, item];
        setImages(updatedData);
        saveData('images', updatedData);
        break;
      case 'files':
        updatedData = [...files, item];
        setFiles(updatedData);
        saveData('files', updatedData);
        break;
    }

    setNewItem({ title: '', content: '', username: '' });
    setShowAddForm(false);
  };

  const handleDelete = (id) => {
    let updatedData;
    switch(activeTab) {
      case 'passwords':
        updatedData = passwords.filter(p => p.id !== id);
        setPasswords(updatedData);
        saveData('passwords', updatedData);
        break;
      case 'emails':
        updatedData = emails.filter(e => e.id !== id);
        setEmails(updatedData);
        saveData('emails', updatedData);
        break;
      case 'documents':
        updatedData = documents.filter(d => d.id !== id);
        setDocuments(updatedData);
        saveData('documents', updatedData);
        break;
      case 'images':
        updatedData = images.filter(i => i.id !== id);
        setImages(updatedData);
        saveData('images', updatedData);
        break;
      case 'files':
        updatedData = files.filter(f => f.id !== id);
        setFiles(updatedData);
        saveData('files', updatedData);
        break;
    }
  };

  const handleMultipleImageUpload = (e) => {
    const uploadedFiles = Array.from(e.target.files);
    
    uploadedFiles.forEach(file => {
      const reader = new FileReader();
      reader.onload = (event) => {
        const item = {
          id: Date.now() + Math.random(),
          title: file.name,
          content: event.target.result,
          fileType: file.type,
          dateAdded: new Date().toLocaleDateString('en-US')
        };

        setImages(prev => {
          const updated = [...prev, item];
          saveData('images', updated);
          return updated;
        });
      };
      reader.readAsDataURL(file);
    });
  };

  const handleFileUpload = (e) => {
    const uploadedFiles = Array.from(e.target.files);
    
    uploadedFiles.forEach(file => {
      const reader = new FileReader();
      reader.onload = (event) => {
        const item = {
          id: Date.now() + Math.random(),
          title: file.name,
          content: event.target.result,
          fileType: file.type,
          fileSize: (file.size / 1024).toFixed(2) + ' KB',
          dateAdded: new Date().toLocaleDateString('en-US')
        };

        setFiles(prev => {
          const updated = [...prev, item];
          saveData('files', updated);
          return updated;
        });
      };
      reader.readAsDataURL(file);
    });
  };

  const handleDocumentUpload = (e) => {
    const file = e.target.files[0];
    if (!file) return;

    const reader = new FileReader();
    reader.onload = (event) => {
      const item = {
        id: Date.now(),
        title: file.name,
        content: event.target.result,
        fileType: file.type,
        dateAdded: new Date().toLocaleDateString('en-US')
      };

      const updatedDocs = [...documents, item];
      setDocuments(updatedDocs);
      saveData('documents', updatedDocs);
    };
    reader.readAsText(file);
  };

  const togglePasswordVisibility = (id) => {
    setShowPassword(prev => ({ ...prev, [id]: !prev[id] }));
  };

  const handleImageClick = (image) => {
    setFullscreenImage(image);
  };

  const handleDownloadImage = () => {
    if (!fullscreenImage) return;
    
    const link = document.createElement('a');
    link.href = fullscreenImage.content;
    link.download = fullscreenImage.title;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const handleDownloadFile = (file) => {
    const link = document.createElement('a');
    link.href = file.content;
    link.download = file.title;
    document.body.appendChild(link);
    link.click();
    document.body.removeChild(link);
  };

  const getFileIcon = (fileType) => {
    if (fileType.includes('pdf')) return 'üìÑ';
    if (fileType.includes('word') || fileType.includes('document')) return 'üìù';
    if (fileType.includes('excel') || fileType.includes('spreadsheet')) return 'üìä';
    if (fileType.includes('powerpoint') || fileType.includes('presentation')) return 'üìΩÔ∏è';
    if (fileType.includes('zip') || fileType.includes('rar')) return 'üóúÔ∏è';
    if (fileType.includes('audio')) return 'üéµ';
    if (fileType.includes('video')) return 'üé¨';
    return 'üìÅ';
  };

  if (isLocked) {
    return (
      <div className="min-h-screen bg-gradient-to-br from-indigo-900 via-purple-900 to-pink-900 flex items-center justify-center p-4">
        <div className="bg-white/10 backdrop-blur-lg rounded-3xl shadow-2xl p-8 w-full max-w-md border border-white/20">
          <div className="text-center mb-8">
            <div className="inline-block p-4 bg-gradient-to-br from-yellow-400 to-orange-500 rounded-full mb-4">
              <Lock className="w-12 h-12 text-white" />
            </div>
            <h1 className="text-3xl font-bold text-white mb-2">Secure Vault</h1>
            <p className="text-gray-300">Enter your password to unlock</p>
            <p className="text-xs text-gray-400 mt-2">Created by JL Magno</p>
          </div>
          
          <div className="space-y-4">
            <div>
              <input
                type="password"
                value={password}
                onChange={(e) => setPassword(e.target.value)}
                onKeyPress={(e) => e.key === 'Enter' && handleUnlock(e)}
                placeholder="Password"
                className="w-full px-4 py-3 bg-white/20 border border-white/30 rounded-xl text-white placeholder-gray-300 focus:outline-none focus:ring-2 focus:ring-yellow-400"
              />
            </div>
            
            {error && (
              <div className="text-red-300 text-sm text-center bg-red-500/20 py-2 rounded-lg">
                {error}
              </div>
            )}
            
            <button
              onClick={handleUnlock}
              className="w-full py-3 bg-gradient-to-r from-yellow-400 to-orange-500 text-white font-semibold rounded-xl hover:from-yellow-500 hover:to-orange-600 transition-all transform hover:scale-105 shadow-lg"
            >
              Unlock Vault
            </button>
          </div>
          
          <div className="text-center mt-6 pt-6 border-t border-white/10">
            <p className="text-xs text-gray-400">¬© 2024 JL Magno. All Rights Reserved.</p>
          </div>
        </div>
      </div>
    );
  }

  const getCurrentData = () => {
    switch(activeTab) {
      case 'passwords': return passwords;
      case 'emails': return emails;
      case 'documents': return documents;
      case 'images': return images;
      case 'files': return files;
      default: return [];
    }
  };

  const tabs = [
    { id: 'passwords', label: 'Passwords', icon: Key, color: 'from-blue-500 to-cyan-500' },
    { id: 'emails', label: 'Emails', icon: Mail, color: 'from-green-500 to-emerald-500' },
    { id: 'documents', label: 'Documents', icon: FileText, color: 'from-purple-500 to-pink-500' },
    { id: 'images', label: 'Images', icon: Image, color: 'from-orange-500 to-red-500' },
    { id: 'files', label: 'Files', icon: File, color: 'from-yellow-500 to-amber-500' }
  ];

  return (
    <div className="min-h-screen bg-gradient-to-br from-slate-900 via-purple-900 to-slate-900 p-4">
      <div className="max-w-6xl mx-auto">
        {/* Header */}
        <div className="bg-white/10 backdrop-blur-lg rounded-2xl shadow-2xl p-6 mb-6 border border-white/20">
          <div className="flex flex-col md:flex-row justify-between items-center gap-4">
            <div className="flex items-center space-x-3">
              <div className="p-3 bg-gradient-to-br from-green-400 to-emerald-500 rounded-xl">
                <Unlock className="w-6 h-6 text-white" />
              </div>
              <div>
                <h1 className="text-2xl font-bold text-white">Secure Vault</h1>
                <p className="text-gray-300 text-sm">Your secure personal storage</p>
              </div>
            </div>
            <div className="flex items-center gap-4">
              <div className="text-center md:text-right">
                <p className="text-xs text-gray-400">Created by</p>
                <p className="text-sm font-semibold text-white">JL Magno</p>
              </div>
              <button
                onClick={handleLock}
                className="flex items-center space-x-2 px-4 py-2 bg-red-500/20 hover:bg-red-500/30 text-red-300 rounded-xl transition-all border border-red-500/30"
              >
                <Lock className="w-4 h-4" />
                <span>Lock</span>
              </button>
            </div>
          </div>
        </div>

        {/* Tabs */}
        <div className="grid grid-cols-2 md:grid-cols-5 gap-4 mb-6">
          {tabs.map(tab => {
            const Icon = tab.icon;
            return (
              <button
                key={tab.id}
                onClick={() => {
                  setActiveTab(tab.id);
                  setShowAddForm(false);
                }}
                className={`p-4 rounded-xl transition-all transform hover:scale-105 ${
                  activeTab === tab.id
                    ? `bg-gradient-to-br ${tab.color} text-white shadow-lg`
                    : 'bg-white/10 text-gray-300 hover:bg-white/20'
                }`}
              >
                <Icon className="w-6 h-6 mx-auto mb-2" />
                <div className="text-sm font-semibold">{tab.label}</div>
              </button>
            );
          })}
        </div>

        {/* Content Area */}
        <div className="bg-white/10 backdrop-blur-lg rounded-2xl shadow-2xl p-6 border border-white/20">
          <div className="flex justify-between items-center mb-6">
            <h2 className="text-xl font-bold text-white">
              {tabs.find(t => t.id === activeTab)?.label}
            </h2>
            <div className="flex space-x-2">
              {activeTab === 'documents' && (
                <label className="flex items-center space-x-2 px-4 py-2 bg-purple-500/20 hover:bg-purple-500/30 text-purple-300 rounded-xl transition-all cursor-pointer border border-purple-500/30">
                  <Plus className="w-4 h-4" />
                  <span>Upload Doc</span>
                  <input type="file" onChange={handleDocumentUpload} className="hidden" accept=".txt,.pdf,.doc,.docx" />
                </label>
              )}
              {activeTab === 'images' && (
                <label className="flex items-center space-x-2 px-4 py-2 bg-orange-500/20 hover:bg-orange-500/30 text-orange-300 rounded-xl transition-all cursor-pointer border border-orange-500/30">
                  <Plus className="w-4 h-4" />
                  <span>Upload Images</span>
                  <input type="file" onChange={handleMultipleImageUpload} className="hidden" accept="image/*" multiple />
                </label>
              )}
              {activeTab === 'files' && (
                <label className="flex items-center space-x-2 px-4 py-2 bg-yellow-500/20 hover:bg-yellow-500/30 text-yellow-300 rounded-xl transition-all cursor-pointer border border-yellow-500/30">
                  <Plus className="w-4 h-4" />
                  <span>Upload Files</span>
                  <input type="file" onChange={handleFileUpload} className="hidden" multiple />
                </label>
              )}
              {(activeTab === 'passwords' || activeTab === 'emails') && (
                <button
                  onClick={() => setShowAddForm(!showAddForm)}
                  className="flex items-center space-x-2 px-4 py-2 bg-green-500/20 hover:bg-green-500/30 text-green-300 rounded-xl transition-all border border-green-500/30"
                >
                  <Plus className="w-4 h-4" />
                  <span>Add New</span>
                </button>
              )}
            </div>
          </div>

          {/* Add Form */}
          {showAddForm && (
            <div className="bg-white/5 rounded-xl p-4 mb-6 border border-white/10">
              <div className="space-y-3">
                <input
                  type="text"
                  placeholder="Name/Title"
                  value={newItem.title}
                  onChange={(e) => setNewItem({...newItem, title: e.target.value})}
                  className="w-full px-4 py-2 bg-white/10 border border-white/20 rounded-lg text-white placeholder-gray-400 focus:outline-none focus:ring-2 focus:ring-blue-400"
                />
                {(activeTab === 'passwords' || activeTab === 'emails') && (
                  <input
                    type="text"
                    placeholder={activeTab === 'passwords' ? 'Username/Email' : 'Email Address'}
                    value={newItem.username}
                    onChange={(e) => setNewItem({...newItem, username: e.target.value})}
                    className="w-full px-4 py-2 bg-white/10 border border-white/20 rounded-lg text-white placeholder-gray-400 focus:outline-none focus:ring-2 focus:ring-blue-400"
                  />
                )}
                <input
                  type="text"
                  placeholder={activeTab === 'passwords' ? 'Password' : 'Content'}
                  value={newItem.content}
                  onChange={(e) => setNewItem({...newItem, content: e.target.value})}
                  className="w-full px-4 py-2 bg-white/10 border border-white/20 rounded-lg text-white placeholder-gray-400 focus:outline-none focus:ring-2 focus:ring-blue-400"
                />
                <div className="flex space-x-2">
                  <button
                    onClick={handleAddItem}
                    className="flex-1 py-2 bg-green-500 hover:bg-green-600 text-white rounded-lg transition-all"
                  >
                    Save
                  </button>
                  <button
                    onClick={() => {
                      setShowAddForm(false);
                      setNewItem({ title: '', content: '', username: '' });
                    }}
                    className="px-4 py-2 bg-red-500/20 hover:bg-red-500/30 text-red-300 rounded-lg transition-all"
                  >
                    Cancel
                  </button>
                </div>
              </div>
            </div>
          )}

          {/* Items List */}
          <div className="space-y-3 max-h-96 overflow-y-auto">
            {getCurrentData().length === 0 ? (
              <div className="text-center py-12 text-gray-400">
                <p>No saved {tabs.find(t => t.id === activeTab)?.label.toLowerCase()} yet</p>
              </div>
            ) : (
              getCurrentData().map(item => (
                <div key={item.id} className="bg-white/5 rounded-xl p-4 border border-white/10 hover:bg-white/10 transition-all">
                  {activeTab === 'images' ? (
                    <div>
                      <div className="flex justify-between items-start mb-2">
                        <h3 className="font-semibold text-white">{item.title}</h3>
                        <button
                          onClick={() => handleDelete(item.id)}
                          className="text-red-400 hover:text-red-300"
                        >
                          <Trash2 className="w-4 h-4" />
                        </button>
                      </div>
                      <img 
                        src={item.content} 
                        alt={item.title} 
                        className="w-full h-48 object-cover rounded-lg cursor-pointer hover:opacity-80 transition-opacity" 
                        onClick={() => handleImageClick(item)}
                      />
                      <p className="text-xs text-gray-400 mt-2">Added: {item.dateAdded}</p>
                    </div>
                  ) : activeTab === 'files' ? (
                    <div>
                      <div className="flex justify-between items-start">
                        <div className="flex items-start space-x-3 flex-1">
                          <div className="text-3xl">{getFileIcon(item.fileType)}</div>
                          <div className="flex-1">
                            <h3 className="font-semibold text-white">{item.title}</h3>
                            <p className="text-xs text-gray-400">Size: {item.fileSize}</p>
                            <p className="text-xs text-gray-400">Added: {item.dateAdded}</p>
                          </div>
                        </div>
                        <div className="flex space-x-2">
                          <button
                            onClick={() => handleDownloadFile(item)}
                            className="text-blue-400 hover:text-blue-300"
                          >
                            <Download className="w-4 h-4" />
                          </button>
                          <button
                            onClick={() => handleDelete(item.id)}
                            className="text-red-400 hover:text-red-300"
                          >
                            <Trash2 className="w-4 h-4" />
                          </button>
                        </div>
                      </div>
                    </div>
                  ) : (
                    <div>
                      <div className="flex justify-between items-start mb-2">
                        <div className="flex-1">
                          <h3 className="font-semibold text-white">{item.title}</h3>
                          {item.username && (
                            <p className="text-sm text-gray-400">{item.username}</p>
                          )}
                        </div>
                        <button
                          onClick={() => handleDelete(item.id)}
                          className="text-red-400 hover:text-red-300"
                        >
                          <Trash2 className="w-4 h-4" />
                        </button>
                      </div>
                      <div className="flex items-center space-x-2">
                        <p className="flex-1 text-gray-300 font-mono text-sm">
                          {activeTab === 'passwords' && !showPassword[item.id] ? '‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢‚Ä¢' : item.content}
                        </p>
                        {activeTab === 'passwords' && (
                          <button
                            onClick={() => togglePasswordVisibility(item.id)}
                            className="text-gray-400 hover:text-white"
                          >
                            {showPassword[item.id] ? <EyeOff className="w-4 h-4" /> : <Eye className="w-4 h-4" />}
                          </button>
                        )}
                      </div>
                      <p className="text-xs text-gray-400 mt-2">Added: {item.dateAdded}</p>
                    </div>
                  )}
                </div>
              ))
            )}
          </div>
        </div>
      </div>

      {/* Fullscreen Image Modal */}
      {fullscreenImage && (
        <div 
          className="fixed inset-0 bg-black/95 z-50 flex items-center justify-center p-4"
          onClick={() => setFullscreenImage(null)}
        >
          <button
            onClick={() => setFullscreenImage(null)}
            className="absolute top-4 right-4 p-3 bg-white/10 hover:bg-white/20 rounded-full text-white transition-all"
          >
            <X className="w-6 h-6" />
          </button>
          
          <button
            onClick={(e) => {
              e.stopPropagation();
              handleDownloadImage();
            }}
            className="absolute bottom-4 right-4 flex items-center space-x-2 px-6 py-3 bg-blue-500 hover:bg-blue-600 rounded-full text-white transition-all shadow-lg"
          >
            <Download className="w-5 h-5" />
            <span>Download</span>
          </button>

          <div className="max-w-7xl max-h-full flex flex-col items-center" onClick={(e) => e.stopPropagation()}>
            <h2 className="text-white text-xl font-semibold mb-4">{fullscreenImage.title}</h2>
            <img 
              src={fullscreenImage.content} 
              alt={fullscreenImage.title} 
              className="max-w-full max-h-[80vh] object-contain rounded-lg shadow-2xl"
            />
          </div>
        </div>
      )}
    </div>
  );
}
